const AWS = require('aws-sdk')

/**
* Constructs a new AwsSpotter Library
* @constructor
* @arg {AwsSvc#constructOpts} construct - The AWS service IAM credentials
*/
function AwsSpotter (constructOps) {
  var opsToConfig = constructOps.keys
  var opsToUpgrade = constructOps.upgrade

  opsToConfig.apiVersion = 'latest'
  opsToConfig.sslEnabled = true

  var stsParams = {
    DurationSeconds: 900,  // 15 minute
    SerialNumber: opsToUpgrade.serialNumber,
    TokenCode: opsToUpgrade.tokenCode
  }

  var sts = new AWS.STS(opsToConfig)

  // Request short term credentials from the STS service
  var req = sts.getSessionToken(stsParams)

  req.on('error', function (err, data) {
    console.log('sts error: ', err)
  })

  req.on('success', function (response) {
    console.log('sts success: ', response.data.hasOwnProperty('Credentials') ?
      response.data : 'BUT missing Credentials!!!')
  })

  req.send()
}
